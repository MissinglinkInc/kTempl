ktempl template engine

普通のテンプレート

syntax

var tmpl1 = new KTempl({template:'templ'});

var result = tmpl1.render({
	name:'僕',
	age:'1',
});

$('#result1').text(result);

result

イテレータ

syntax

var tmpl2 = new KTempl({template:'templ'});

var result2 = tmpl2.render([
{
	name:'僕',
	age:'5',
},
{
	name:'あなた',
	age:'12',
},
]);

$('#result2').html(result2);

result
